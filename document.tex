	\section*{Subscript T Comparison Test}
	
	Below are several ways to typeset the subscript capital T.
	
	\bigskip
	
	1. Default
	
	\[
	x_T
	\]
	
	\bigskip
	
	2. Using text font
	
	\[
	x_{\text{T}}
	\]
	
	\bigskip
	
	3. Using roman math font
	
	\[
	x_{\mathrm{T}}
	\]
	
	\bigskip
	
	4. Using scriptstyle
	
	\[
	x_{\scriptstyle T}
	\]
	
	\bigskip
	
	5. Using scriptscriptstyle
	
	\[
	x_{\scriptscriptstyle T}
	\]
	
	\bigskip
	
	6. Negative thin space adjustment
	
	\[
	x_{\!T}
	\]
	
	\bigskip
	
	7. Lowercase comparison
	
	\[
	x_t
	\]
	
	\bigskip
	
	\section*{Inline Comparison}
	
	Default: $x_T$  
	
	Text: $x_{\text{T}}$  
	
	Roman: $x_{\mathrm{T}}$  
	
	Scriptstyle: $x_{\scriptstyle T}$  
	
	Scriptscriptstyle: $x_{\scriptscriptstyle T}$  
	
	Negative space: $x_{\!T}$  
	
	Lowercase: $x_t$
